function buildChart(id, history) { const ctx = document.getElementById(id); new Chart(ctx, { type: 'line', data: { labels: history.map(h => h.d), datasets: [{ data: history.map(h => h.v), borderColor: '#00FF9C', backgroundColor: 'rgba(0, 255, 156, 0.1)', fill: true, tension: 0.4, pointRadius: 6, pointBackgroundColor: '#00FF9C', pointHitRadius: 40 }] }, options: { responsive: true, maintainAspectRatio: false, layout: { padding: { top: 30 // Extra space at the top so the tooltip doesn't overlap the line } }, interaction: { mode: 'index', intersect: false }, plugins: { legend: { display: false }, tooltip: { enabled: true, position: 'nearest', caretSize: 0, // Removes the "arrow" pointing to the dot backgroundColor: 'rgba(0, 0, 0, 0.8)', titleFont: { size: 14 }, bodyFont: { size: 22, weight: 'bold' }, // Huge font for the weight bodyAlign: 'center', titleAlign: 'center', padding: 10, displayColors: false, // Custom positioning: This forces the tooltip to stay at the top external: function(context) { let tooltipEl = document.getElementById('chartjs-tooltip'); if (!tooltipEl) { tooltipEl = document.createElement('div'); tooltipEl.id = 'chartjs-tooltip'; document.body.appendChild(tooltipEl); } }, callbacks: { label: (ctx) => `${ctx.raw.toLocaleString()} lbs` } } }, scales: { x: { display: false }, y: { display: false, grace: '20%' // Adds 20% "headroom" above the highest data point } } } }); }
